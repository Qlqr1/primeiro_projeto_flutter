import 'package:flutter/material.dart';

void main() {
  runApp(
    const MaterialApp(home: PaginaInicial(), debugShowCheckedModeBanner: false),
  );
}

// ── Modelo ──────────────────────────────────────────────────────────────────

class Aluno {
  final String nome;
  final String telefone;
  final String matricula;

  const Aluno({
    required this.nome,
    required this.telefone,
    required this.matricula,
  });
}

// ── Tela principal ───────────────────────────────────────────────────────────

class PaginaInicial extends StatefulWidget {
  const PaginaInicial({super.key});

  @override
  State<PaginaInicial> createState() => _PaginaInicialState();
}

class _PaginaInicialState extends State<PaginaInicial> {
  final List<Aluno> _alunos = [];

  Future<void> _abrirFormulario() async {
    final aluno = await Navigator.push<Aluno>(
      context,
      MaterialPageRoute(builder: (_) => const FormularioCadastro()),
    );
    if (aluno != null) {
      setState(() => _alunos.add(aluno));
    }
  }

  Future<void> _confirmarExclusao(int index) async {
    final confirmado = await showDialog<bool>(
      context: context,
      builder: (ctx) => AlertDialog(
        title: const Text('Excluir aluno'),
        content: Text(
          'Tem certeza que quer excluir o aluno ${_alunos[index].nome}?',
        ),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(ctx, false),
            child: const Text('Cancelar'),
          ),
          TextButton(
            onPressed: () => Navigator.pop(ctx, true),
            child: const Text('Excluir', style: TextStyle(color: Colors.red)),
          ),
        ],
      ),
    );
    if (confirmado == true) {
      setState(() => _alunos.removeAt(index));
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Olá 3F <3'),
        leading: const Icon(Icons.menu),
        centerTitle: true,
        backgroundColor: Colors.blue,
      ),
      body: _alunos.isEmpty
          ? const Center(
              child: Text(
                'Nenhum aluno cadastrado.\nToque em + para adicionar.',
                textAlign: TextAlign.center,
                style: TextStyle(color: Colors.grey),
              ),
            )
          : ListView.builder(
              padding: const EdgeInsets.all(8),
              itemCount: _alunos.length,
              itemBuilder: (_, index) => _AlunoCard(
                aluno: _alunos[index],
                onDelete: () => _confirmarExclusao(index),
              ),
            ),
      floatingActionButton: FloatingActionButton(
        onPressed: _abrirFormulario,
        child: const Icon(Icons.add),
      ),
    );
  }
}

// ── Card do aluno ────────────────────────────────────────────────────────────

class _AlunoCard extends StatelessWidget {
  const _AlunoCard({required this.aluno, required this.onDelete});

  final Aluno aluno;
  final VoidCallback onDelete;

  @override
  Widget build(BuildContext context) {
    return Card(
      elevation: 4,
      margin: const EdgeInsets.symmetric(horizontal: 8, vertical: 6),
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
      // InkWell deixa o card clicável com o efeito de ripple do Material
      child: InkWell(
        borderRadius: BorderRadius.circular(10),
        onTap: () => Navigator.push(
          context,
          MaterialPageRoute(builder: (_) => DetalhesAluno(aluno: aluno)),
        ),
        child: Padding(
          padding: const EdgeInsets.all(16),
          child: Row(
            children: [
              const Icon(Icons.people_sharp, color: Colors.blue, size: 36),
              const SizedBox(width: 16),
              Expanded(
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Text(
                      aluno.nome,
                      style: const TextStyle(
                        fontWeight: FontWeight.bold,
                        fontSize: 16,
                      ),
                    ),
                    const SizedBox(height: 4),
                    Text('Telefone: ${aluno.telefone}'),
                    Text('Matrícula: ${aluno.matricula}'),
                  ],
                ),
              ),
              IconButton(
                icon: const Icon(Icons.delete_outline, color: Colors.redAccent),
                onPressed: onDelete,
              ),
            ],
          ),
        ),
      ),
    );
  }
}

// ── Tela de detalhes do aluno ────────────────────────────────────────────────

class DetalhesAluno extends StatelessWidget {
  const DetalhesAluno({super.key, required this.aluno});

  final Aluno aluno;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Detalhes do Aluno'),
        centerTitle: true,
        backgroundColor: Colors.blue,
        // O botão de voltar é adicionado automaticamente pelo Flutter quando
        // existe uma rota anterior na pilha de navegação. Não precisa criar
        // manualmente — o AppBar já detecta e exibe a seta de voltar sozinho.
      ),
      body: Padding(
        padding: const EdgeInsets.all(24),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Avatar com inicial do nome
            Center(
              child: CircleAvatar(
                radius: 48,
                backgroundColor: Colors.blue,
                child: Text(
                  aluno.nome.isNotEmpty ? aluno.nome[0].toUpperCase() : '?',
                  style: const TextStyle(fontSize: 40, color: Colors.white),
                ),
              ),
            ),
            const SizedBox(height: 32),
            _InfoItem(icone: Icons.person, label: 'Nome', valor: aluno.nome),
            const Divider(),
            _InfoItem(
              icone: Icons.phone,
              label: 'Telefone',
              valor: aluno.telefone,
            ),
            const Divider(),
            _InfoItem(
              icone: Icons.badge,
              label: 'Matrícula',
              valor: aluno.matricula,
            ),
            const Divider(),
            const SizedBox(height: 32),
            SizedBox(
              width: double.infinity,
              child: OutlinedButton.icon(
                onPressed: () => Navigator.pop(context),
                icon: const Icon(Icons.arrow_back),
                label: const Text('Voltar'),
                style: OutlinedButton.styleFrom(
                  minimumSize: const Size.fromHeight(48),
                ),
              ),
            ),
          ],
        ),
      ),
    );
  }
}

// ── Item de informação ───────────────────────────────────────────────────────

class _InfoItem extends StatelessWidget {
  const _InfoItem({
    required this.icone,
    required this.label,
    required this.valor,
  });

  final IconData icone;
  final String label;
  final String valor;

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.symmetric(vertical: 12),
      child: Row(
        children: [
          Icon(icone, color: Colors.blue, size: 28),
          const SizedBox(width: 16),
          Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(
                label,
                style: const TextStyle(fontSize: 12, color: Colors.grey),
              ),
              Text(
                valor,
                style: const TextStyle(
                  fontSize: 16,
                  fontWeight: FontWeight.w500,
                ),
              ),
            ],
          ),
        ],
      ),
    );
  }
}

// ── Formulário de cadastro ───────────────────────────────────────────────────

class FormularioCadastro extends StatefulWidget {
  const FormularioCadastro({super.key});

  @override
  State<FormularioCadastro> createState() => _FormularioCadastroState();
}

class _FormularioCadastroState extends State<FormularioCadastro> {
  final _formKey = GlobalKey<FormState>();
  final _nomeCtrl = TextEditingController();
  final _telefoneCtrl = TextEditingController();
  final _matriculaCtrl = TextEditingController();

  @override
  void dispose() {
    _nomeCtrl.dispose();
    _telefoneCtrl.dispose();
    _matriculaCtrl.dispose();
    super.dispose();
  }

  void _salvar() {
    if (_formKey.currentState!.validate()) {
      Navigator.pop(
        context,
        Aluno(
          nome: _nomeCtrl.text.trim(),
          telefone: _telefoneCtrl.text.trim(),
          matricula: _matriculaCtrl.text.trim(),
        ),
      );
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Formulário'),
        centerTitle: true,
        backgroundColor: Colors.blue,
      ),
      body: Form(
        key: _formKey,
        child: ListView(
          padding: const EdgeInsets.all(24),
          children: [
            _CampoTexto(
              label: 'Nome',
              controller: _nomeCtrl,
              validator: (v) =>
                  (v == null || v.trim().isEmpty) ? 'Informe o nome' : null,
            ),
            const SizedBox(height: 16),
            _CampoTexto(
              label: 'Telefone',
              controller: _telefoneCtrl,
              keyboardType: TextInputType.phone,
              validator: (v) =>
                  (v == null || v.trim().isEmpty) ? 'Informe o telefone' : null,
            ),
            const SizedBox(height: 16),
            _CampoTexto(
              label: 'Matrícula',
              controller: _matriculaCtrl,
              validator: (v) => (v == null || v.trim().isEmpty)
                  ? 'Informe a matrícula'
                  : null,
            ),
            const SizedBox(height: 32),
            ElevatedButton(
              onPressed: _salvar,
              style: ElevatedButton.styleFrom(
                minimumSize: const Size.fromHeight(48),
              ),
              child: const Text('Salvar'),
            ),
          ],
        ),
      ),
    );
  }
}

// ── Campo de texto reutilizável ──────────────────────────────────────────────

class _CampoTexto extends StatelessWidget {
  const _CampoTexto({
    required this.label,
    required this.controller,
    this.validator,
    this.keyboardType,
  });

  final String label;
  final TextEditingController controller;
  final String? Function(String?)? validator;
  final TextInputType? keyboardType;

  @override
  Widget build(BuildContext context) {
    return TextFormField(
      controller: controller,
      keyboardType: keyboardType,
      validator: validator,
      decoration: InputDecoration(
        labelText: label,
        border: const OutlineInputBorder(),
      ),
    );
  }
}
